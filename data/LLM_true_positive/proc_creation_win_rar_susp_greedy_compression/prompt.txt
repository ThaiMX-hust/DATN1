Sigma_rule:
title: Suspicious Greedy Compression Using Rar.EXE
id: afe52666-401e-4a02-b4ff-5d128990b8cb

tags:
    - attack.execution

    product: windows
    category: process_creation
detection:

            - ' ?:\Users\Public\'

    condition: 1 of selection_opt_* and all of selection_cli_*
